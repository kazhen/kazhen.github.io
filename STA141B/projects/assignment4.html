def ndb_search(term, maxrow = '500', key = "wNUzBo4FYzkpl9cVsO9TksqvAD8BOCxZrkTN6zpd") :
    """
    This function makes a search request based on the food name
    Input: Food name, max row, key
    Output: List of results or empty list if term is invalid
    
    """
    
    #food url and request
    food_url = "https://api.nal.usda.gov/ndb/search/?format=json&q={}&max={}&sort=r&offset=0&api_key={}".format(term, maxrow, key)
    request = requests.get(food_url)
    rr = request.json()
    
    try:
        #invalid term
        rr['errors']
        return []
    except:
        #return list
        return rr[u'list'][u'item']

def ndb_nums(path = '../Desktop/fresh.csv'):
    """
    This function opens a csv file and searches all the foods for the NDB numbers
    Input: Path of file
    Output: Dataframe with NDB numbers appended to csv file
    """
    
    #open file 
    fruitfile = pd.read_csv(path)
    nbd_df = pd.DataFrame()

    for i in range(0, len(fruitfile)) :
        #searches through entire file and replaces underscores with spaces
        foodname = fruitfile.ix[i, 5]
        foodname = foodname.replace("_", " ")
        x = ndb_search(foodname)
        x = pd.DataFrame(x)
        #remove all brand foods
        x = x[x.ds != 'BL']
        x = x.reset_index(drop = True)
    
        flag1 = 0
        j = 0
        while (flag1 == 0) :
            #search through all SR foods
            if (j == len(x)) :
                flag1 = 1
                nbd = 0
                break
        
            foodstr = str(x.ix[j,2])
            foodstr = foodstr.split(", ")
        
            k = 0
            flag = 0
            while (flag == 0) :
                #search through name 
                if (k == len(foodstr)):
                    flag = 1
                    break
                
                if (foodstr[k] == "raw") :
                    #if raw find, store NBD number
                    flag1 = 1
                    flag = 1
                    nbd = str(x.ix[j, 3])
                else :
                    k += 1
        
            j += 1
    
        #append to ndb dataframe
        nbd_df = nbd_df.append({'NDB': nbd}, ignore_index = True) 

    #concatenate to csv file
    new_df = pd.concat([fruitfile, nbd_df], axis = 1, join_axes=[fruitfile.index])
    return new_df

def ndb_report(num, key = "wNUzBo4FYzkpl9cVsO9TksqvAD8BOCxZrkTN6zpd") :
    """
    This function gets the nutrition info from NDB number
    Input: NDB number and key
    Output: List of nutrition
    """
    
    #uses url and calls request
    breed_uri = "https://api.nal.usda.gov/ndb/V2/reports?ndbno={}&type=b&format=json&api_key={}".format(num,key)
    request = Request(breed_uri)
    response = urlopen(request)
    rr = response.read()
    js = json.loads(rr)
    nut = js[u'foods'][0]['food']['nutrients']
    
    return nut
